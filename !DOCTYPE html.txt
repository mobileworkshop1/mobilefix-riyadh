    <script>
        // نفس الكود السابق (الدخول + الحجوزات + لوحة التحكم)
        function showLoginModal() { document.getElementById('loginModal').classList.remove('hidden'); }
        function hideLoginModal() { document.getElementById('loginModal').classList.add('hidden'); }

        let currentUser = null;

        function login() {
            const email = document.getElementById('email').value;
            const pass = document.getElementById('password').value;
            const users = {
                'admin@mobilefix.com': { pass: 'admin123', role: 'admin', name: 'الإدمن' },
                'customer@demo.com': { pass: '1234', role: 'customer', name: 'أحمد العميل' }
            };
            if (users[email] && users[email].pass === pass) {
                currentUser = { email, ...users[email] };
                hideLoginModal();
                alert(`تم تسجيل الدخول بنجاح! مرحبا ${currentUser.name}`);
                if (currentUser.role === 'admin') {
                    document.getElementById('dashboard').classList.remove('hidden');
                    renderBookings();
                }
            } else {
                alert('بيانات خاطئة! جرب admin@mobilefix.com / admin123');
            }
        }

        function logout() {
            currentUser = null;
            document.getElementById('dashboard').classList.add('hidden');
        }

        let bookings = JSON.parse(localStorage.getItem('bookings')) || [
            { id: 1, customer: 'أحمد', car: 'ألمانية', service: 'تغيير زيت', date: '2026-04-22', status: 'قيد الانتظار' },
            { id: 2, customer: 'خالد', car: 'أمريكية', service: 'بطارية', date: '2026-04-23', status: 'مكتمل' }
        ];

        function saveBookings() { localStorage.setItem('bookings', JSON.stringify(bookings)); }

        function renderBookings() {
            const tbody = document.getElementById('bookingsTable');
            tbody.innerHTML = '';
            bookings.forEach((b, index) => {
                const row = document.createElement('tr');
                row.className = 'border-b hover:bg-gray-50';
                row.innerHTML = `
                    <td class="p-4">${b.id}</td>
                    <td class="p-4">${b.customer}</td>
                    <td class="p-4">${b.car}</td>
                    <td class="p-4">${b.service}</td>
                    <td class="p-4">${b.date}</td>
                    <td class="p-4"><span class="px-4 py-1 rounded-full ${b.status==='مكتمل'?'bg-green-100 text-green-700':'bg-yellow-100 text-yellow-700'}">${b.status}</span></td>
                    <td class="p-4">
                        <button onclick="updateStatus(${index})" class="text-green-600 mx-2">✅</button>
                        <button onclick="deleteBooking(${index})" class="text-red-600">🗑️</button>
                    </td>
                `;
                tbody.appendChild(row);
            });
        }

        function updateStatus(i) {
            bookings[i].status = bookings[i].status === 'قيد الانتظار' ? 'مكتمل' : 'قيد الانتظار';
            saveBookings();
            renderBookings();
        }

        function deleteBooking(i) {
            if (confirm('حذف الحجز؟')) {
                bookings.splice(i, 1);
                saveBookings();
                renderBookings();
            }
        }

        function handleBooking(e) {
            e.preventDefault();
            const newBooking = {
                id: bookings.length + 1,
                customer: currentUser ? currentUser.name : 'عميل جديد',
                car: document.getElementById('carType').value,
                service: document.getElementById('service').value,
                date: document.getElementById('date').value,
                status: 'قيد الانتظار'
            };
            bookings.push(newBooking);
            saveBookings();
            alert('✅ تم تسجيل الحجز بنجاح! سيتواصل معك الفريق قريباً');
            e.target.reset();
            if (currentUser && currentUser.role === 'admin') renderBookings();
        }

        function showBooking() {
            document.getElementById('booking').scrollIntoView({ behavior: 'smooth' });
        }
    </script>
